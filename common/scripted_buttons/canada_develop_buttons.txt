canada_develop_button_n_qb = {
	name = canada_develop_button_n_qb
	desc = canada_develop_button_n_qb_desc

	visible = { always = yes }

	possible = {
		average_sol >= 10
		
		GDP >= 100000
		hidden_trigger = {
			NOT = { has_variable = qb_develop }
		}
	}
	
	effect = {
		set_global_variable = enable_north_quebec_dev
		set_variable = {
			name = qb_develop
		}
	}
	ai_chance = {
		base = 1
	}
}
